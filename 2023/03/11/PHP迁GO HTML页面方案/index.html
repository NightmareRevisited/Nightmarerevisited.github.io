PHP模块迁GO前端页面方案

背景

业务的同步数据页面（PC端、SEO）使用php的smarty模板框架进行数据注入，但是在go中并没有任何现成模板框架支持php的samrty，所以迁go后如何基于php模块现有模板实现同步页面是一个关键问题。

方案

在Go语言中，除了标准库自带的html/template包之外，还有种类繁多的第三方模板引擎库，这些库大多来自其他语言的经验继承。但是很遗憾，并没有类似smarty模板语法的库。

编号 方案 学习成本 rd开发量 fe开发量 旧代码复用程度 维护成本 上线时间
1 服务端渲染（ssr) fe学习成本巨大，除了nodejs ssr渲染外，还需要学习服务器维护管理、持续集成等服务端开发需要的知识，但这显然和fe技术栈没有任何交集 数据同步页面的后端代码需要改动适配 巨大 现有数据同步页面完全无法复用，需要改成支持ssr的页面 fe需要自己维护ssr集群 由于fe自身还有业务开发任务，考虑到学习ssr、页面改造、服务器集群搭建及管理维护，很难在短时间PHP完全迁移GO模块
2 fe舍弃smarty模板转go模板 同样，fe学习成本巨大，需要从0学习go的任意一个模板引擎，搭建goserver本地调试也需要从0学习 无 巨大 fe代码完全无法复用 无 纯人力堆积方案，同样很难在短时间内做到完全迁移
3 smarty模板预编译成go template然后go直接使用 rd需要学习smarty模板知识，fe新页面开发没有任何改动，仍然使用smarty即可 中等 无 完全可以复用 无 即插即用，实现后立刻就可以做到任意功能在go模块开发上线

考虑到人力成本、学习成本和预期上线问题，最终选择方案3。

package main

import (
	"bytes"
	"io"
	"io/fs"
	"os"
	"path/filepath"
	"regexp"
	"strings"
)

var (
	smartyTplPath = "smarty模板路径"
	tempTplPath   = "中转路径"
)

func main() {
	//php原始smarty模板转中间模板
	smarty2Temp()

	//中间模板编译go template
	temp2GoTemplate()
}

// temp2GoTemplate
// @Description: 中间模板转smarty
func temp2GoTemplate() {
	conf := &smarty.Config{
		LeftDelimiter:  "<%",
		RightDelimiter: "%>",
		TemplateDir:    tempTplPath,
		CompileDir:     "webroot/aiqifu/views",
		Ignores:        []string{"inc/*"},
		UcaseField:     false,
		CompileOnline:  false,
		Glob:           false,
		AutoRoot:       false,
		Mode:           smartyTypes.Smarty,
	}
	fInst, _ := smarty.New(conf)
	_, err := fInst.CompileAll()
	if err != nil {
		panic(err)
	}
}

// smarty2Temp
// @Description: 原始smarty模板转中间模板
func smarty2Temp() {
	var tplPathList []string
	err := filepath.Walk(smartyTplPath, func(path string, info fs.FileInfo, err error) error {
		if info.IsDir() || !tools.StringEndWith(path, ".tpl") {
			return nil
		}

		path = path[len(smartyTplPath)+1:]
		tplPathList = append(tplPathList, path)
		return nil
	})
	if err != nil {
		panic(err)
	}

	smartyTplBlock := regexp.MustCompile("<%.*?%>")

	variableComp := regexp.MustCompile(`\$[\w.]+`)
	forLoopStartComp := regexp.MustCompile(`<% *foreach *\$[\w.]+ *as *(\$[\w.]+) *(=>)? *(\$[\w.]+)?.*%>`)
	forLoopEndComp := regexp.MustCompile(`<% */ *foreach *%>`)
	assignmentComp := regexp.MustCompile(`<% *(\$[\w.]+)* *=[^=].*%>`)

	for _, tplPath := range tplPathList {
		smartyTplFile := filepath.Join(smartyTplPath, tplPath)
		tempTplFile := filepath.Join(tempTplPath, tplPath)
		smartyFd, err := os.OpenFile(smartyTplFile, os.O_RDONLY, 0666)
		if err != nil {
			panic(err)
		}

		fileContent := &bytes.Buffer{}
		_, err = io.Copy(fileContent, smartyFd)
		if err != nil {
			panic(err)
		}
		fileString := fileContent.String()

		var localVariableStack = [][]string{
			{"$smarty"}, //smarty保留变量
		}

		smartyBlock := smartyTplBlock.FindAllString(fileString, -1)
		for _, block := range smartyBlock {
			//判断是否为赋值语句
			assignmentMatch := assignmentComp.FindStringSubmatch(block)
			if len(assignmentMatch) > 0 {
				localVariableStack[len(localVariableStack)-1] = append(localVariableStack[len(localVariableStack)-1], assignmentMatch[1])
			}

			//判断是否为foreach
			forLoopStartMatch := forLoopStartComp.FindStringSubmatch(block)
			if len(forLoopStartMatch) > 0 {
				localVariableStack = append(localVariableStack, []string{})
				localVariableStack[len(localVariableStack)-1] = append(localVariableStack[len(localVariableStack)-1], forLoopStartMatch[1])
				if forLoopStartMatch[3] != "" {
					localVariableStack[len(localVariableStack)-1] = append(localVariableStack[len(localVariableStack)-1], forLoopStartMatch[3])
				}
			}

			//判断是否结束foreach
			forLoopEndMatch := forLoopEndComp.FindStringSubmatch(block)
			if len(forLoopEndMatch) > 0 {
				localVariableStack = localVariableStack[:len(localVariableStack)-1]
			}

			var newBlock = strings.ReplaceAll(block, "'", "\"")
			newBlock = strings.ReplaceAll(newBlock, "$smarty.now", "\"\"|now")

			variableList := variableComp.FindAllString(block, -1)
			for _, variable := range variableList {
				if isLocalVariable(localVariableStack, variable) {
					continue
				}

				newBlock = strings.ReplaceAll(newBlock, variable, variable[:1]+"ROOT."+variable[1:])
			}

			if newBlock != block {
				fileString = strings.ReplaceAll(fileString, block, newBlock)
			}
		}

		//摘除所有strip 和 json_encode
		rule1 := regexp.MustCompile(`<%/?strip%>`)
		rule2 := regexp.MustCompile(`\| *json_encode *`)

		fileString = rule1.ReplaceAllString(fileString, "")
		fileString = rule2.ReplaceAllString(fileString, "")

		tempFd, err := util.CreateFile(tempTplFile)
		if err != nil {
			panic(err)
		}
		_, err = tempFd.WriteString(fileString)
		if err != nil {
			panic(err)
		}
	}
}

// isLocalVariable
// @Description: 在作用域中逐层查找变量
// @param variableStack 作用域（函数栈）
// @param variable
// @return bool
func isLocalVariable(variableStack [][]string, variable string) bool {
	for _, floor := range variableStack {
		for _, variablePrefix := range floor {
			if len(variablePrefix) > len(variable) {
				continue
			}

			if variable[:len(variablePrefix)] == variablePrefix {
				return true
			}
		}
	}

	return false
}
