    类成员访问运算符->重载Tips

    2021-02-18

    概述

    c++中对成员访问运算符->的重载属于特殊的重载

    1. ->是一个unary operator，如果试图在->重载函数中添加参数，编译会直接报错；
    2. ->重载的是指向对象的指针，对于一个class的实例obj，&obj->attr 默认是从obj的指针中去访问成员的，重载后从->操作符返回的指针中去访问成员。和动态语言的properties get/set有很大差异，不要混为一谈。
    3. 为了防止不可预测的内部赋值引发事故，通常将->和->*的重载函数定义为const来禁止对内部成员的修改。

    Tips

    1. 和其他运算符重载不同，->的重载在类内部是不生效的
    #include <iostream>
#include <vector>
using namespace std;

class Obj {
    static int i, j;
public:
    void f() const { cout << i++ << endl; }
    void g() const { cout << j++ << endl; }
};

int Obj::i = 10;
int Obj::j = 12;

class ObjContainer {
    vector<Obj*> a;
public:
    void add(Obj* obj)
    {
        a.push_back(obj);  
    }
    friend class SmartPointer;
};

class SmartPointer {
    ObjContainer oc;
public:
    int index;

    SmartPointer(ObjContainer& objc)
    {
        oc = objc;
        index = 0;
    }
    bool operator++() 
    {
//         this->f();
//         尝试访问Obj instance的成员函数f会引发编译报错
        if(this->index >= oc.a.size() - 1) return false;
        if(oc.a[++index] == 0) return false;
        return true;
    }
    bool operator++(int)
    {
        return operator++();
    }

    Obj* operator->() const
    {
        if(!oc.a[index])
        {
            cout << "Zero value";
            return (Obj*)0;
        }
        return oc.a[index];
    }
};

int main() {
    const int sz = 10;
    Obj o[sz];
    ObjContainer oc;
    for(int i = 0; i < sz; i++)
    {
        oc.add(&o[i]);
    }
    SmartPointer sp(oc); 
    do {
        sp->f(); 
        sp->g();
//        cout << sp->index << endl;
//        如果在类外部通过->访问sp的成员会引发编译报错，此时->访问的对象是Obj而非SmartPointer
    } while(sp++);
    return 0;
}
